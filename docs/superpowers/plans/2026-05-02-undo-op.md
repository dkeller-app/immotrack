# UNDO-OP Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Ajouter un mécanisme d'annulation multi-niveau (Ctrl+Z + FAB + toast inline) couvrant toutes les écritures DB d'ImmoTrack, avec libellés explicites pour les ~11 suppressions critiques.

**Architecture:** Stack RAM 20 niveaux de snapshots `structuredClone(DB)` capturés au boot puis après chaque `saveDB()` réussi. Hook minimal dans `saveDB()` pour pousser l'ancienne snapshot avec le bon libellé (explicite via `_undoOp(label, fn)` ou anonyme via auto-debounce 500 ms). Restore en place via `Object.assign(DB, snapshot)` puis re-render `go(currentPage)`.

**Tech Stack:** Vanilla JS (ES2020+), monolith `index.html`, IndexedDB pour photos (intouché), Drive sync via `_drvSync` (intouché), `structuredClone` natif, pas de dépendance externe.

**Spec source:** [docs/subjects/UNDO-OP.md](../../subjects/UNDO-OP.md)

```js
/* ═══ UNDO-OP — v14.21+ (session 2026-05-02) ════════════════════════════════
   Stack RAM 20 niveaux d'annulation avec snapshot complet de DB.
   Cf docs/subjects/UNDO-OP.md.

   FLOW :
   1. Au boot (post-loadDB), _undoLastSnapshot capture l'état initial.
   2. À chaque saveDB() réussi, l'ancienne _undoLastSnapshot est poussée dans
      la stack (= état d'avant la modif courante), puis _undoLastSnapshot est
      mis à jour avec le nouvel état.
   3. _undoOp(label, fn) set _undoNextLabel pendant l'exécution de fn() pour
      que le hook saveDB sache quel libellé utiliser.
   4. Sans label, on auto-debounce 500ms : tous les saveDB en rafale fusionnent
      en une seule entrée "Modification" qui pointe vers la 1re snapshot
      pré-rafale (préservée dans _undoDebouncedPrev).
   5. _undoUndo() pop la dernière entrée, wipe DB en place, Object.assign
      restore, saveDB push Drive, go(currentPage) re-render.
*/

const UNDO_MAX = 20;
const UNDO_DEBOUNCE_MS = 500;
let _undoStack = [];
let _undoNextLabel = null;
let _undoDebounceTimer = null;
let _undoDebouncedPrev = null;
let _undoLastSnapshot = null;
let _undoSuppressCapture = false;

function _undoSnapshot() {
  return structuredClone(DB);
}

function _undoOp(label, fn) {
  const prev = _undoNextLabel;
  _undoNextLabel = label;
  try { fn(); }
  finally { _undoNextLabel = prev; }
}

function _undoOnSaveDB() {
  if (_undoSuppressCapture) return;
  // _undoLastSnapshot représente l'état d'AVANT la modif qu'on est en train
  // de committer (capturée à la fin du saveDB précédent ou au boot).
  if (!_undoLastSnapshot) return; // 1er saveDB du boot : rien à pousser
  const prevState = _undoLastSnapshot;
  if (_undoNextLabel) {
    // Op labellée : push immédiat, annule un éventuel debounce en cours
    clearTimeout(_undoDebounceTimer);
    _undoDebouncedPrev = null;
    _undoStack.push({ label: _undoNextLabel, snapshot: prevState, ts: Date.now() });
    if (_undoStack.length > UNDO_MAX) _undoStack.shift();
    _undoUIRefresh();
  } else {
    // Op anonyme : préserver la 1re snapshot pré-rafale, push différé
    if (!_undoDebouncedPrev) _undoDebouncedPrev = prevState;
    clearTimeout(_undoDebounceTimer);
    _undoDebounceTimer = setTimeout(() => {
      _undoStack.push({ label: 'Modification', snapshot: _undoDebouncedPrev, ts: Date.now() });
      if (_undoStack.length > UNDO_MAX) _undoStack.shift();
      _undoDebouncedPrev = null;
      _undoUIRefresh();
    }, UNDO_DEBOUNCE_MS);
  }
}

function _undoOnSaveDBSuccess() {
  // Appelé À LA FIN d'un saveDB réussi : capture le nouvel état comme prev
  // pour la prochaine modif.
  if (_undoSuppressCapture) return;
  _undoLastSnapshot = _undoSnapshot();
}

function _undoUndo() {
  if (typeof _appReadOnly !== 'undefined' && _appReadOnly) {
    if (typeof _showReadOnlyAlert === 'function') _showReadOnlyAlert();
    return;
  }
  if (!_undoStack.length) {
    showToast('Rien à annuler', '', 2000);
    return;
  }
  const entry = _undoStack.pop();
  _undoSuppressCapture = true;
  // Wipe + restore DB en place (préserve la ref globale `DB`)
  Object.keys(DB).forEach(k => delete DB[k]);
  Object.assign(DB, entry.snapshot);
  // Refresh la "last snapshot" pour qu'elle reflète l'état restauré
  _undoLastSnapshot = _undoSnapshot();
  saveDB();
  _undoSuppressCapture = false;
  // Re-render page courante
  if (typeof go === 'function' && typeof currentPage !== 'undefined' && currentPage) {
    try { go(currentPage); } catch(e) { console.warn('[undo] re-render fail:', e); }
  }
  // Toast confirmation (avec hint Drive folder si suppression)
  const driveHint = entry.label.indexOf('Suppression') === 0
    ? ' · Le dossier Drive reste à la corbeille (récup 30j)'
    : '';
  showToast('✅ Annulé : ' + entry.label + driveHint, 'ok', 5000);
  _undoUIRefresh();
}

function _undoClear(reason) {
  if (!_undoStack.length && !_undoDebouncedPrev) return;
  _undoStack = [];
  clearTimeout(_undoDebounceTimer);
  _undoDebouncedPrev = null;
  _undoUIRefresh();
  if (reason === 'drive_pull') {
    showToast('↻ Synchro Drive — historique d\'annulation effacé', '', 3500);
  }
}

function _undoUIRefresh() {
  const fab = document.getElementById('fab-undo');
  if (!fab) return;
  if (_undoStack.length) {
    fab.style.display = '';
    const last = _undoStack[_undoStack.length - 1];
    fab.title = '↶ Annuler : ' + last.label;
  } else {
    fab.style.display = 'none';
    fab.title = '↶ Annuler';
  }
}

function _inEditableField(target) {
  if (!target) return false;
  const tag = target.tagName;
  if (tag === 'INPUT' || tag === 'TEXTAREA' || tag === 'SELECT') return true;
  if (target.isContentEditable) return true;
  return false;
}
/* ═══ FIN UNDO-OP cœur ════════════════════════════════════════════════════ */

```

```js
function _undoUIInit() {
  // FAB bouton (injecté une fois au boot)
  if (!document.getElementById('fab-undo')) {
    const fab = document.createElement('button');
    fab.id = 'fab-undo';
    fab.type = 'button';
    fab.setAttribute('aria-label', 'Annuler la dernière opération');
    fab.title = '↶ Annuler';
    fab.textContent = '↶';
    fab.addEventListener('click', _undoUndo);
    document.body.appendChild(fab);
  }
  // Ctrl+Z global avec guard editable field
  window.addEventListener('keydown', (e) => {
    if (!(e.ctrlKey || e.metaKey)) return;
    if (e.key !== 'z' && e.key !== 'Z') return;
    if (e.shiftKey) return; // réservé pour redo futur
    if (_inEditableField(e.target)) return; // laisser le navigateur faire son undo natif
    e.preventDefault();
    _undoUndo();
  });
  _undoUIRefresh();
}
```
